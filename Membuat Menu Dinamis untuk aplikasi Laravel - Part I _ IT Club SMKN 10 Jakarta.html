CREATE TABLE `menu` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `id_induk` int(11) NOT NULL,
  `nama` varchar(100) COLLATE utf8_unicode_ci NOT NULL,
  `icon` varchar(50) COLLATE utf8_unicode_ci NOT NULL,
  `url` varchar(100) COLLATE utf8_unicode_ci NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=101 DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

function get_menu_child($parent=0){
  $menu = \App\Menu::where('id_induk',$parent)->get();
  $parent = \App\Menu::where('id',$parent)->first();
  ?>
  <li class="treeview">
    <a href="{{url($parent->url)}}">
      <i class="fa {{$parent->icon}}"></i>
      <span>{{$parent->nama}}</span>
      @if(sizeof($menu)>0)
      <i class="fa fa-angle-left pull-right"></i>
      @endif
    </a>
    @if(sizeof($menu)>0)
    <ul class="treeview-menu">
      <?php
      foreach ($menu as $key) {
        get_menu_child($key->id);
      }
      ?>
    </ul>
    @endif
  </li>
<?php } ?>
